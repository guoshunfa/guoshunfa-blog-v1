class Solution {
    public int search(int[] nums, int target) {
        int left = 0;
        int right = nums.length-1;
        while(left <= right ) {
            int mi = left + (right-left) /2;
            if (nums[mi]== target) {
                return mi;
            } else if (nums[mi] > target)  {
                right = mi-1;
            } else if (nums[mi] < target)  {
                left = mi+1;
            } 
        }
        return -1;
    }
}
